# Golang排序算法

### 冒泡排序

```go
func bubbleSort(nums []int) []int {
	n := len(nums)
	for i := 0; i < n - 1; i++ {
		exchange := false
		for j := 0; j < n-i-1; j++ {
			if nums[j] > nums[j+1] { //两两比较，前面比后面大
				nums[j], nums[j+1] = nums[j+1], nums[j] //交换
				exchange = true
			}
		}
		if !exchange { //如果已经拍好序，跳出循环
			return nums
		}
	}
	return nums
}
```



### 插入排序

```go
func insertSort(nums []int) []int {
	n := len(nums)
	for i := 1; i < n; i++ {
		tmp := nums[i]
		j := i - 1
		for j >= 0 && nums[j] > tmp { //左边比右边大
			nums[j+1] = nums[j] //右移1位
			j--                 //扫描前一个数
		}
		nums[j+1] = tmp //添加到小于它的数的右边
	}
	return nums
}
```



### 选择排序

```go
func selectSort(nums []int) []int {
    n := len(nums)
    for i:=0;i<n-1;i++{
        imin:=i
        for j:=i+1;j<n;j++{
            if nums[j]<nums[imin]{
                imin = j
            }
        }
        if imin!=i{
            nums[i],nums[imin]=nums[imin],nums[i]
        }
    }
    return nums
}
```



### 基数排序

```go
//只适用于数组全为非负数
func radixSort(li []int) []int{
    max_num := li[0]  //找出数组中最大的数
    for i := 0; i < len(li); i++ {
        if max_num < li[i] {
            max_num = li[i]
        }
    }
    //把每个位比作一个桶
    for j := 0; j < len(strconv.Itoa(max_num)); j++ {
        bin := make([][]int, 10)//创建桶
        //入桶
        for k := 0; k < len(li); k++ {
            n := li[k] / int(math.Pow(10, float64(j))) % 10
            bin[n] = append(bin[n], li[k])
        }
        //出桶
        m := 0
        for p := 0; p < len(bin); p++ {
            for q := 0; q < len(bin[p]); q++ {
                li[m] = bin[p][q]
                m++
            }
        }
    }
    return li
}
```



### 堆排序

```go
func heapSort(nums []int) []int {
    lens := len(nums) - 1
    for i := lens/2; i >= 0; i -- { // 建堆 O(n) lens/2后面都是叶子节点，不需要向下调整
        down(nums, i, lens)
    }
    for j := lens; j >= 1; j -- {//堆排序（升序）:堆顶(最大值)交换到末尾
        nums[0], nums[j] = nums[j], nums[0]
        lens --
        down(nums, 0, lens)
    }
    return nums
}

//O(logn)大根堆，如果堆顶节点小于叶子，向下调整 
func down(nums []int, i, lens int) {
    max := i  //比较当前节点的子节点，2i+1和2i+2
    if i<<1+1 <= lens && nums[i<<1+1] > nums[max] {
        max = i<<1+1
    }  
    if i<<1+2 <= lens && nums[i<<1+2] > nums[max] {
        max = i<<1 + 2
    }
    if max != i {
        nums[max], nums[i] = nums[i], nums[max]
        down(nums, max, lens)
    }
}
```



### 快速排序

```go
func quickSort(nums []int, l, r int) {
	if l >= r {
		return
	}
	i, j := l, r
	nums[i], nums[(i+j)>>1] = nums[(i+j)>>1], nums[i]
	pivot := nums[i]
	for i < j {
		for i < j && pivot <= nums[j] {
			j--
		}
		nums[i] = nums[j]
		for i < j && nums[i] <= pivot {
			i++
		}
		nums[j] = nums[i]
	}
	nums[i] = pivot
	quickSort(nums, l, i-1)
	quickSort(nums, i+1, r)
}
```



### 归并排序

```go
//合并两段有序数组
func merge(nums []int, left, mid, right int) {
	i, j := left, mid+1
	tmp := []int{}
	for i <= mid && j <= right {
		if nums[i] < nums[j] {
			tmp = append(tmp, nums[i])
			i++
		} else {
			tmp = append(tmp, nums[j])
			j++
		}
	}
	if i <= mid {
		tmp = append(tmp, nums[i:mid+1]...)
	} else {
		tmp = append(tmp, nums[j:right+1]...)
	}
	copy(nums[left:right+1], tmp)
}

func mergeSort(nums []int, left, right int) {
	if left < right {
		mid := (left + right) / 2
		mergeSort(nums, left, mid)
		mergeSort(nums, mid+1, right)
		merge(nums, left, mid, right)
	}
}
```
